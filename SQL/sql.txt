(
   NAMECAR             VARCHAR2(255 Char)   NOT NULL,
   ACTIONDURATION      NUMBER(10)           NOT NULL,
   BRAND               VARCHAR2(255 Char),
   HAVEPOSITIONDEVICE  VARCHAR2(255 Char),
   HAVEPOWERSTEERING   VARCHAR2(255 Char),
   NUMBERPLATE         NUMBER(10),
   OTHER               VARCHAR2(255 Char),
   YEAROFMANUFACTURE   NUMBER(10)
)

ALTER TABLE CAR
   ADD PRIMARY KEY (NAMECAR) 
   USING INDEX TABLESPACE SYSAUX;
